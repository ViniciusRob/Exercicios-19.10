void gera(int vetor[]){
  int l, num;
  for (l = 0; l < 20; l++){
    num = rand() % 50;
    vetor[l] = num;
  }
}

void bubbleSort(int vetor[], int size){
  int l, d, aux;
  for(l = 1; l < size; l++){

    for(d = 0; d < size - l; d++){

      if(vetor[d] < vetor[d+1]){
        aux = vetor[d];
        vetor[d] = vetor[d+1];
        vetor [d+1] = aux;
      }
    }
  }
}
